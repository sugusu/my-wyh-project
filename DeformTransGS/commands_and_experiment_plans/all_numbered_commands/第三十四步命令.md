def isolate_state(
    df: pd.DataFrame,
    state: str,
) -> pd.DataFrame:
    out = (
        df.loc[
            df["state"].eq(state)
        ]
        .copy()
        .reset_index(drop=True)
    )

    if out.empty:
        raise RuntimeError(
            f"No rows for state={state}"
        )

    states_found = out["state"].unique()

    if (
        len(states_found) != 1
        or states_found[0] != state
    ):
        raise RuntimeError(
            "State isolation failed: "
            f"requested={state}, "
            f"found={states_found}"
        )

    return out

def compare_state_reproduction(
    r4_df: pd.DataFrame,
    p0_df: pd.DataFrame,
    state: str,
) -> dict[str, float]:
    key = [
        "state",
        "cell_id",
    ]

    r4_state = isolate_state(
        r4_df,
        state,
    )

    p0_state = isolate_state(
        p0_df,
        state,
    )

    assert_unique_key(
        f"R4/{state}",
        r4_state,
    )

    assert_unique_key(
        f"P0/{state}",
        p0_state,
    )

    r4_col = "R_cell"

    if r4_col not in r4_state.columns:
        raise KeyError(
            f"R4 missing {r4_col}"
        )

    if r4_col not in p0_state.columns:
        raise KeyError(
            f"P0 missing {r4_col}"
        )

    merged = p0_state[
        key + [r4_col]
    ].merge(
        r4_state[
            key + [r4_col]
        ],
        on=key,
        how="inner",
        validate="one_to_one",
        suffixes=(
            "_new",
            "_r4",
        ),
    )

    if merged.shape[0] == 0:
        raise RuntimeError(
            f"No aligned rows for {state}"
        )

    diff = np.abs(
        merged["R_cell_new"].to_numpy(
            dtype=np.float64
        )
        -
        merged["R_cell_r4"].to_numpy(
            dtype=np.float64
        )
    )

    return {
        "state": state,
        "n": int(diff.size),
        "median_diff": float(
            np.median(diff)
        ),
        "p95_diff": float(
            np.quantile(diff, 0.95)
        ),
        "p99_diff": float(
            np.quantile(diff, 0.99)
        ),
        "max_diff": float(
            np.max(diff)
        ),
    }

def aggregate_cross_camera_r(
    values: np.ndarray,
) -> float:
    values = np.asarray(
        values,

    values = values[
        np.isfinite(values)

    if values.size < 2:

import torch


def tensor_max_abs_diff(
    a: torch.Tensor,
    b: torch.Tensor,
) -> float:
    a = a.detach().cpu().to(
        torch.float64
    )

    b = b.detach().cpu().to(
        torch.float64
    )

    if a.shape != b.shape:
        raise ValueError(
            f"shape mismatch: "
            f"{a.shape} vs {b.shape}"
        )

    return float(
        torch.max(
            torch.abs(a-b)
        ).item()
    )
