from dataclasses import dataclass

import torch


@dataclass
class GaussianTensors:
    xyz: torch.Tensor
    scale: torch.Tensor
    rotation: torch.Tensor
    tau: torch.Tensor
    color: torch.Tensor
    material_id: torch.Tensor


def validate_gaussians(g: GaussianTensors) -> None:
    n = g.xyz.shape[0]

    if g.xyz.shape != (n, 3):
        raise ValueError("xyz shape invalid")

    if g.scale.shape != (n, 3):
        raise ValueError("scale shape invalid")

    if g.rotation.shape != (n, 4):
        raise ValueError("rotation shape invalid")

    if g.tau.reshape(-1).shape[0] != n:
        raise ValueError("tau shape invalid")

    if g.color.shape != (n, 3):
        raise ValueError("color shape invalid")

    if g.material_id.reshape(-1).shape[0] != n:
        raise ValueError("material_id shape invalid")

    for name, tensor in {
        "xyz": g.xyz,
        "scale": g.scale,
        "rotation": g.rotation,
        "tau": g.tau,
        "color": g.color,
    }.items():
        if not torch.isfinite(tensor).all():
            raise ValueError(f"{name} contains non-finite values")


def clone_gaussians(
    reference: GaussianTensors,
    parent_indices: torch.Tensor,
    mode: str,
) -> GaussianTensors:
    """
    Exact duplicate clone.

    mode:
        naive
        opacity_corrected

    Clone geometry must be IDENTICAL
    to the selected parent geometry.

    No random offset.
    No scale change.
    No rotation change.
    """

    validate_gaussians(reference)

    parent_indices = parent_indices.to(
        device=reference.xyz.device,
        dtype=torch.long,
    ).reshape(-1)

    if parent_indices.numel() == 0:
        raise ValueError("No clone parents selected")

    if parent_indices.min() < 0:
        raise ValueError("Negative parent index")

    if parent_indices.max() >= reference.xyz.shape[0]:
        raise ValueError("Parent index out of range")

    xyz = reference.xyz.clone()
    scale = reference.scale.clone()
    rotation = reference.rotation.clone()
    tau = reference.tau.clone()
    color = reference.color.clone()
    material_id = reference.material_id.clone()

    clone_xyz = reference.xyz[parent_indices].clone()
    clone_scale = reference.scale[parent_indices].clone()
    clone_rotation = reference.rotation[parent_indices].clone()
    clone_tau = reference.tau[parent_indices].clone()
    clone_color = reference.color[parent_indices].clone()
    clone_material_id = reference.material_id[parent_indices].clone()

    if mode == "naive":
        # Original-style opacity-preserving clone.
        # Parent and clone keep the original tau.
        pass

    elif mode == "opacity_corrected":
        # Revising Densification clone correction
        # expressed in tau parameterization.
        #
        # (1-o_new)^2 = 1-o_old
        # o = 1-exp(-tau)
        # => tau_new = tau_old / 2

        corrected_tau = (
            reference.tau[parent_indices].clone()
            / 2.0
        )

        tau[parent_indices] = corrected_tau
        clone_tau = corrected_tau.clone()

    else:
        raise ValueError(
            f"Unknown clone mode: {mode}"
        )

    out = GaussianTensors(
        xyz=torch.cat([xyz, clone_xyz], dim=0),
        scale=torch.cat([scale, clone_scale], dim=0),
        rotation=torch.cat([rotation, clone_rotation], dim=0),
        tau=torch.cat([tau, clone_tau], dim=0),
        color=torch.cat([color, clone_color], dim=0),
        material_id=torch.cat(
            [material_id, clone_material_id],
            dim=0,
        ),
    )

    validate_gaussians(out)

    return out

    g: GaussianTensors,
    material_u: torch.Tensor,
    material_v: torch.Tensor,
    deformation,
) -> tuple[GaussianTensors, torch.Tensor]:

    Transport parent and clones by material identity.

    Multiple Gaussians may share one material_id.
    They MUST receive identical phi, F, Js
    before covariance transport.
    """

    mid = g.material_id.long()

    if mid.min() < 0:
        raise ValueError("Negative material id")

    if mid.max() >= material_u.shape[0]:
        raise ValueError("Material id out of range")

    u = material_u[mid]
    v = material_v[mid]

        xyz=g.xyz,
        u=u,
