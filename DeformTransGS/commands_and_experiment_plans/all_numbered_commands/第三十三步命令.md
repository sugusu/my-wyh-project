def covariance_from_scale_rotation(
    scale: torch.Tensor,
    rotation: torch.Tensor,
) -> torch.Tensor:
    if scale.ndim != 2 or scale.shape[1] != 3:
        raise ValueError(
            f"scale shape invalid: {scale.shape}"
        )

    if rotation.ndim != 2 or rotation.shape[1] != 4:
        raise ValueError(
            f"rotation shape invalid: {rotation.shape}"
        )

    if torch.any(scale <= 0):
        raise ValueError(
            "scale must be activated and positive"
        )

    R = quaternion_wxyz_to_matrix(rotation)

    D = torch.diag_embed(scale.square())

    Sigma = (
        R
        @ D
        @ R.transpose(1, 2)
    )

    return 0.5 * (
        Sigma

    )

def polar_rotation(
    F: torch.Tensor,
) -> torch.Tensor:
    """
    F: [N,3,3]
    return proper rotation [N,3,3]
    """

    U, _, Vh = torch.linalg.svd(F)

    R = U @ Vh

    det = torch.linalg.det(R)

    neg = det < 0

    if torch.any(neg):
        U = U.clone()
        U[neg, :, -1] *= -1.0
        R = U @ Vh

    det_after = torch.linalg.det(R)

    if torch.max(
        torch.abs(det_after - 1.0)
    ) > 1e-5:
        raise RuntimeError(
            "polar rotation determinant invalid"
        )

    ortho = (
        R
        @ R.transpose(1, 2)
    )

    eye = torch.eye(
        3,
        device=F.device,
        dtype=F.dtype,
    ).unsqueeze(0)

    if torch.max(
        torch.abs(ortho - eye)
    ) > 1e-5:
        raise RuntimeError(
            "polar rotation is not orthogonal"
        )

    return R

def orthonormal_deformed_tangent(
    F: torch.Tensor,
    e_u: torch.Tensor,
    e_v: torch.Tensor,
) -> torch.Tensor:
    tu = torch.einsum(
        "nij,j->ni",
        F,
        e_u,
    )

    tv = torch.einsum(
        "nij,j->ni",
        F,
        e_v,
    )

    e1 = tu / torch.linalg.norm(
        tu,
        dim=1,
        keepdim=True,
    ).clamp_min(1e-12)

    tv_ortho = (
        tv
        -
        torch.sum(
            tv * e1,
            dim=1,
            keepdim=True,
        )
        * e1
    )

    e2 = tv_ortho / torch.linalg.norm(
        tv_ortho,
        dim=1,
        keepdim=True,
    ).clamp_min(1e-12)

    return torch.stack(
        [e1, e2],
        dim=2,
    )  # [N,3,2]
