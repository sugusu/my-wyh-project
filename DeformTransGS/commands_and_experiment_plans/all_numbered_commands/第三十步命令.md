from __future__ import annotations

import torch


def project_points_cuda_exact(
    xyz: torch.Tensor,
    camera,
) -> dict[str, torch.Tensor]:
    """
    Reproduce the current local TSGS CUDA rasterizer
    transformPoint4x4 + perspective divide + ndc2Pix.

    xyz:
        [N, 3], world-space points.

    IMPORTANT:
    camera.full_proj_transform already contains
    world-view and projection transforms.

    DO NOT multiply world_view_transform again.
    """

    if xyz.ndim != 2 or xyz.shape[1] != 3:
        raise ValueError(
            f"xyz must have shape [N,3], got {tuple(xyz.shape)}"
        )

    xyz = xyz.to(
        device=camera.full_proj_transform.device,
        dtype=camera.full_proj_transform.dtype,
    )

    ones = torch.ones(
        (xyz.shape[0], 1),
        device=xyz.device,
        dtype=xyz.dtype,
    )

    xyz_h = torch.cat([xyz, ones], dim=1)  # [N,4]

    M = camera.full_proj_transform

    if M.shape != (4, 4):
        raise ValueError(
            f"full_proj_transform must be [4,4], got {tuple(M.shape)}"
        )

    # CUDA transformPoint4x4 layout:
    #
    # p_hom.x =
    #   M_flat[0]  * x +
    #   M_flat[4]  * y +
    #   M_flat[8]  * z +
    #   M_flat[12]
    #
    # With the tensor layout used by TSGS Camera,
    # this corresponds to row-vector multiplication:
    #
    # xyz_h @ M
    #
    # NOT M @ xyz_h.T
    # NOT xyz_h @ world_view_transform @ M

    p_hom = xyz_h @ M  # [N,4]

    inv_w = 1.0 / (p_hom[:, 3] + 1e-7)

    ndc_x = p_hom[:, 0] * inv_w
    ndc_y = p_hom[:, 1] * inv_w
    ndc_z = p_hom[:, 2] * inv_w

    W = int(camera.image_width)
    H = int(camera.image_height)

    # Exact CUDA ndc2Pix:
    # ((v + 1) * S - 1) * 0.5
    #
    # NO y flip.

    pixel_x = ((ndc_x + 1.0) * W - 1.0) * 0.5
    pixel_y = ((ndc_y + 1.0) * H - 1.0) * 0.5

    in_frame = (
        (pixel_x >= 0.0)
        & (pixel_x < W)
        & (pixel_y >= 0.0)
        & (pixel_y < H)
    )

    finite = (
        torch.isfinite(pixel_x)
        & torch.isfinite(pixel_y)
        & torch.isfinite(ndc_z)
        & torch.isfinite(p_hom[:, 3])
    )

    return {
        "pixel_x": pixel_x,
        "pixel_y": pixel_y,
        "ndc_z": ndc_z,
        "hom_w": p_hom[:, 3],
        "in_frame": in_frame,
        "finite": finite,
    }

import numpy as np


def bilinear_sample(
    image: np.ndarray,
    x: np.ndarray,
    y: np.ndarray,
) -> np.ndarray:
    """
    image: [H,W] float array
    x: horizontal / column coordinate
    y: vertical / row coordinate
    """

    image = np.asarray(image, dtype=np.float64)
    x = np.asarray(x, dtype=np.float64).reshape(-1)
    y = np.asarray(y, dtype=np.float64).reshape(-1)

    if x.shape != y.shape:
        raise ValueError("x and y must have identical shape")

    H, W = image.shape

    valid = (
        np.isfinite(x)
        & np.isfinite(y)
        & (x >= 0.0)
        & (x < W - 1)
        & (y >= 0.0)
        & (y < H - 1)
    )

    out = np.full(x.shape, np.nan, dtype=np.float64)

    xv = x[valid]
    yv = y[valid]

    x0 = np.floor(xv).astype(np.int64)
    y0 = np.floor(yv).astype(np.int64)
    x1 = x0 + 1
    y1 = y0 + 1

    wx = xv - x0
    wy = yv - y0

    Ia = image[y0, x0]
    Ib = image[y0, x1]
    Ic = image[y1, x0]
    Id = image[y1, x1]

    out[valid] = (
        (1.0 - wx) * (1.0 - wy) * Ia
        + wx * (1.0 - wy) * Ib
        + (1.0 - wx) * wy * Ic
        + wx * wy * Id
    )

    return out

def make_cell_quadrature(
    u_low: float,
    u_high: float,
    v_low: float,
    v_high: float,
    q: int,
) -> tuple[np.ndarray, np.ndarray]:
    if q < 2:
        raise ValueError("q must be >= 2")

    # Mid-cell tensor-product samples.
    # Avoid repeatedly placing samples exactly
    # on shared cell boundaries.

    u_edges = np.linspace(u_low, u_high, q + 1)
    v_edges = np.linspace(v_low, v_high, q + 1)

    u_samples = 0.5 * (u_edges[:-1] + u_edges[1:])
    v_samples = 0.5 * (v_edges[:-1] + v_edges[1:])

    uu, vv = np.meshgrid(
        u_samples,
        v_samples,
        indexing="ij",
    )

    return uu.reshape(-1), vv.reshape(-1)

def interpolate_uniform_response(
    q_query: float,
    q_ref: np.ndarray,
    r_ref: np.ndarray,
) -> float:
    q_ref = np.asarray(q_ref, dtype=np.float64)
    r_ref = np.asarray(r_ref, dtype=np.float64)

    order = np.argsort(q_ref)

    q_sorted = q_ref[order]
    r_sorted = r_ref[order]

    if q_query < q_sorted[0] or q_query > q_sorted[-1]:
        return np.nan

    return float(
        np.interp(
            q_query,
            q_sorted,
            r_sorted,
        )
    )

def paired_signflip_test(
    delta: np.ndarray,
    n_perm: int = 10000,
    seed: int = 20260713,
) -> float:
    delta = np.asarray(
        delta,
        dtype=np.float64,
    )

    delta = delta[np.isfinite(delta)]

    if delta.size == 0:
        raise ValueError("No finite paired samples")

    rng = np.random.default_rng(seed)

    observed = abs(np.mean(delta))

    count = 0

    for _ in range(n_perm):
        signs = rng.choice(
            np.array([-1.0, 1.0]),
            size=delta.size,
            replace=True,
        )

        stat = abs(
            np.mean(delta * signs)
        )

        if stat >= observed:
            count += 1

    return (count + 1) / (n_perm + 1)
