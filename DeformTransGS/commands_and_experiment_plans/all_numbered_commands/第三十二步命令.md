from dataclasses import dataclass

import torch


@dataclass
class GaussianState:
    xyz: torch.Tensor          # [N, 3], world coordinates
    scale: torch.Tensor        # [N, 3], ACTIVATED positive scale
    rotation: torch.Tensor     # [N, 4], quaternion wxyz
    tau: torch.Tensor          # [N] or [N,1]

def validate_state(g: GaussianState) -> None:
    n = g.xyz.shape[0]

    expected = {
        "xyz": (n, 3),
        "scale": (n, 3),
        "rotation": (n, 4),
        "color": (n, 3),
    }

    for name, shape in expected.items():
        tensor = getattr(g, name)

        if tuple(tensor.shape) != shape:
            raise ValueError(
                f"{name}: expected {shape}, "
                f"got {tuple(tensor.shape)}"
            )

        if not torch.isfinite(tensor).all():
            raise ValueError(
                f"{name} contains non-finite values"
            )

        raise ValueError("tau length mismatch")

    if g.material_id.reshape(-1).shape[0] != n:
        raise ValueError("material_id length mismatch")

    if torch.any(g.scale <= 0):
        raise ValueError(
            "scale must be activated positive scale"
        )

    q_norm = torch.linalg.norm(
        g.rotation,
        dim=1,
    )

    if torch.max(torch.abs(q_norm - 1.0)) > 1e-5:

def quaternion_wxyz_to_matrix(
    q: torch.Tensor,
) -> torch.Tensor:
    """
    q: [N,4], quaternion order w,x,y,z
    """

    q = q / torch.linalg.norm(
        q,
        dim=1,
        keepdim=True,
    ).clamp_min(1e-12)

    w, x, y, z = q.unbind(dim=1)

    R = torch.empty(
        (q.shape[0], 3, 3),
        device=q.device,
        dtype=q.dtype,
    )

    R[:, 0, 0] = 1 - 2 * (y*y + z*z)
    R[:, 0, 1] = 2 * (x*y - w*z)
    R[:, 0, 2] = 2 * (x*z + w*y)

    R[:, 1, 0] = 2 * (x*y + w*z)
    R[:, 1, 1] = 1 - 2 * (x*x + z*z)
    R[:, 1, 2] = 2 * (y*z - w*x)

    R[:, 2, 0] = 2 * (x*z - w*y)
    R[:, 2, 1] = 2 * (y*z + w*x)
    R[:, 2, 2] = 1 - 2 * (x*x + y*y)

    return R

def covariance_from_scale_rotation(
    scale: torch.Tensor,
    rotation: torch.Tensor,
) -> torch.Tensor:
    """
    scale: [N,3], activated positive standard deviations
    rotation: [N,4], normalized wxyz quaternion

    returns Sigma [N,3,3]
    """

    if torch.any(scale <= 0):
        raise ValueError(
            "Activated scale must be positive"
        )

    R = quaternion_wxyz_to_matrix(rotation)

    D = torch.diag_embed(scale * scale)

    Sigma = R @ D @ R.transpose(1, 2)

    Sigma = 0.5 * (
        Sigma + Sigma.transpose(1, 2)
    )

    return Sigma

def transport_covariance(
    scale: torch.Tensor,
    rotation: torch.Tensor,
    F: torch.Tensor,
) -> torch.Tensor:
    """
    Return transported covariance Sigma_def.
    """

    Sigma = covariance_from_scale_rotation(
        scale=scale,
        rotation=rotation,
    )

    if F.shape != Sigma.shape:
        raise ValueError(
            f"F shape {F.shape} != Sigma shape {Sigma.shape}"
        )

    Sigma_def = (
        F
        @ Sigma
        @ F.transpose(1, 2)
    )

    Sigma_def = 0.5 * (
        Sigma_def
        + Sigma_def.transpose(1, 2)
    )

    if not torch.isfinite(Sigma_def).all():
        raise ValueError(
            "Sigma_def contains non-finite values"
        )

    return Sigma_def

九、关键代码：Covariance → Scale/Rotation

使用 symmetric eigendecomposition：

def covariance_to_scale_rotation(
    Sigma: torch.Tensor,
) -> tuple[torch.Tensor, torch.Tensor]:
    """
    Sigma: [N,3,3], symmetric positive definite.

    Return:
        scale [N,3]
        quaternion wxyz [N,4]
    """

    Sigma = 0.5 * (
        Sigma + Sigma.transpose(1, 2)
    )

    eigvals, eigvecs = torch.linalg.eigh(Sigma)

    eigvals = torch.clamp(
        eigvals,
        min=1e-12,
    )

    scale = torch.sqrt(eigvals)

    R = eigvecs

    det = torch.linalg.det(R)

    neg = det < 0

    if torch.any(neg):
        R = R.clone()
        R[neg, :, 0] *= -1.0

    rotation = rotation_matrix_to_quaternion_wxyz(R)

    rotation = rotation / torch.linalg.norm(
        rotation,
        dim=1,
        keepdim=True,
    ).clamp_min(1e-12)

    return scale, rotation

def transport_gaussians_validated(
    g: GaussianState,
    deformation,
    material_u: torch.Tensor,
    material_v: torch.Tensor,
) -> tuple[GaussianState, torch.Tensor, torch.Tensor]:
    """
    Apply material deformation and covariance transport.

    Returns:
        deformed Gaussian state
        F [N,3,3]
        Js [N]
    """

    validate_state(g)

    mid = g.material_id.reshape(-1).long()

    u = material_u[mid]
    v = material_v[mid]

    xyz_def, F, Js = deformation(
        xyz=g.xyz,
        u=u,
        v=v,
    )

    if xyz_def.shape != g.xyz.shape:
        raise ValueError("xyz_def shape mismatch")

    if F.shape != (
        g.xyz.shape[0],
        3,
        3,
    ):
        raise ValueError("F shape mismatch")

    if Js.reshape(-1).shape[0] != g.xyz.shape[0]:
        raise ValueError("Js shape mismatch")

    Sigma_def = transport_covariance(
        scale=g.scale,
        rotation=g.rotation,
        F=F,
    )

    scale_def, rotation_def = (
        covariance_to_scale_rotation(
            Sigma_def
        )
    )

    out = GaussianState(
        xyz=xyz_def,
        scale=scale_def,
        rotation=rotation_def,
        tau=g.tau.clone(),
        color=g.color.clone(),
        material_id=g.material_id.clone(),
    )

    validate_state(out)

    return out, F, Js
