from __future__ import annotations

from pathlib import Path

import numpy as np
import pandas as pd


class HistoricalAlphaProvider:
    def __init__(
        self,
        manifest_csv: Path,
    ) -> None:
        self.df = pd.read_csv(
            manifest_csv
        )

        required = {
            "state",
            "camera_id",
            "recovered_path",
            "actual_sha256",
            "manifest_sha256",
            "sha_match",
        }

        missing = (
            required
            - set(self.df.columns)
        )

        if missing:
            raise KeyError(
                f"Missing columns: {missing}"
            )

        key = [
            "state",
            "camera_id",
        ]

        if self.df.duplicated(
            key
        ).any():
            raise RuntimeError(
                "Duplicated alpha keys"
            )

        if not self.df[
            "sha_match"
        ].astype(bool).all():
            raise RuntimeError(
                "Historical alpha SHA mismatch"
            )

        self.lookup = {
            (
                str(row.state),
                str(row.camera_id),
            ): Path(row.recovered_path)
            for row in self.df.itertuples()
        }

    def load(
        self,
        state: str,
        camera_id: str,
    ) -> np.ndarray:
        key = (
            str(state),
            str(camera_id),
        )

        if key not in self.lookup:
            raise KeyError(
                f"Missing historical alpha: {key}"
            )

        alpha = np.load(
            self.lookup[key]
        )

        alpha = np.asarray(
            alpha,
            dtype=np.float64,
        )

        if alpha.ndim != 2:
            raise ValueError(
                f"Alpha must be [H,W], "
                f"got {alpha.shape}"
            )

        if not np.isfinite(alpha).all():
            raise ValueError(
                f"Non-finite alpha: {key}"
            )

        return alpha

def compare_alpha(
    a_hist,
    a_current,
):
    a = np.asarray(
        a_hist,
        dtype=np.float64,
    )

    b = np.asarray(
        a_current,
        dtype=np.float64,
    )

    if a.shape != b.shape:
        raise ValueError(
            f"Shape mismatch: "
            f"{a.shape} vs {b.shape}"
        )

    d = np.abs(a-b)

    return {
        "mae": float(
            d.mean()
        ),
        "median": float(
            np.median(d)
        ),
        "p95": float(
            np.quantile(d, 0.95)
        ),
        "p99": float(
            np.quantile(d, 0.99)
        ),
        "max": float(
            d.max()
        ),
        "fraction_gt_1e-6": float(
            np.mean(d >1e-6)
        ),
        "fraction_gt_1e-4": float(
            np.mean(d >1e-4)
        ),
        "fraction_gt_1e-2": float(

from __future__ import annotations

import hashlib

import numpy as np
import torch


def tensor_content_sha256(
    value,
) -> str:
    if isinstance(
        value,
        torch.Tensor,
    ):
        array = (
            value.detach()
            .cpu()
            .contiguous()
            .numpy()
        )
    else:
        array = np.asarray(
            value
        )

    array = np.ascontiguousarray(
        array
    )

    return hashlib.sha256(
        array.tobytes()
    ).hexdigest()
