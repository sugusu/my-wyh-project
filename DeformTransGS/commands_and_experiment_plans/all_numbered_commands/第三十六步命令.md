from __future__ import annotations

import ast
import hashlib
from pathlib import Path


def normalized_function_hash(
    source_path: Path,
    function_name: str,
) -> str | None:
    source = source_path.read_text(
        encoding="utf-8"
    )

    tree = ast.parse(source)

    for node in ast.walk(tree):
        if (
            isinstance(node, ast.FunctionDef)
            and node.name == function_name
        ):
            text = ast.unparse(node)

            return hashlib.sha256(
                text.encode("utf-8")
            ).hexdigest()

    return None

import numpy as np


def compare_arrays_nan_aware(
    a,
    b,
    atol: float,
) -> dict:
    a = np.asarray(a)
    b = np.asarray(b)

    if a.shape != b.shape:
        return {
            "shape_equal": False,
            "equal": False,
            "reason": "shape_mismatch",
            "max_abs_diff": np.nan,
        }

    if (
        np.issubdtype(a.dtype, np.bool_)
        or np.issubdtype(b.dtype, np.bool_)
    ):
        mismatch = (
            a.astype(bool)
            != b.astype(bool)
        )

        return {
            "shape_equal": True,
            "equal": not mismatch.any(),
            "reason": (
                "equal"
                if not mismatch.any()
                else "bool_mismatch"
            ),
            "max_abs_diff": float(
                mismatch.any()
            ),
        }

    af = np.asarray(
        a,
        dtype=np.float64,
    )

    bf = np.asarray(
        b,
        dtype=np.float64,
    )

    nan_a = np.isnan(af)
    nan_b = np.isnan(bf)

    if not np.array_equal(
        nan_a,
        nan_b,
    ):
        return {
            "shape_equal": True,
            "equal": False,
            "reason": "nan_mask_mismatch",
            "max_abs_diff": np.nan,
        }

    posinf_a = np.isposinf(af)
    posinf_b = np.isposinf(bf)

    neginf_a = np.isneginf(af)
    neginf_b = np.isneginf(bf)

    if (
        not np.array_equal(
            posinf_a,
            posinf_b,
        )
        or
        not np.array_equal(
            neginf_a,
            neginf_b,
        )
    ):
        return {
            "shape_equal": True,
            "equal": False,
            "reason": "inf_mask_mismatch",
            "max_abs_diff": np.nan,
        }

    finite = (
        np.isfinite(af)
        & np.isfinite(bf)
    )

    if not finite.any():
        # All positions have matching NaN / Inf masks.
        return {
            "shape_equal": True,
            "equal": True,
            "reason": "matching_nonfinite_only",
            "max_abs_diff": 0.0,
        }

    diff = np.abs(
        af[finite] - bf[finite]
    )

    max_diff = float(diff.max())

    return {
        "shape_equal": True,
        "equal": max_diff <= atol,
        "reason": (
            "equal"
            if max_diff <= atol
            else "finite_value_mismatch"
        ),
        "max_abs_diff": max_diff,
    }

from __future__ import annotations

import hashlib

import numpy as np
import torch


def tensor_sha256(
    tensor: torch.Tensor,
) -> str:
    array = (
        tensor
        .detach()
        .cpu()
        .contiguous()
        .numpy()
    )

    return hashlib.sha256(
        array.tobytes()
    ).hexdigest()


@torch.no_grad()
def repeat_render(
    render_once,
    n_repeat: int = 20,
):
    outputs = []

    for repeat_id in range(n_repeat):
        torch.cuda.synchronize()

        alpha = render_once()

        torch.cuda.synchronize()

        alpha_np = (
            alpha
            .detach()
            .cpu()
            .to(torch.float64)
            .numpy()
            .copy()
        )

        outputs.append(alpha_np)

    return outputs
