def polar_rotation(
    F: torch.Tensor,
) -> torch.Tensor:
    U, _, Vh = torch.linalg.svd(F)

    R = U @ Vh

    det = torch.linalg.det(R)

    neg = det < 0

    if torch.any(neg):
        U = U.clone()
        U[neg, :, -1] *= -1.0
        R = U @ Vh

    eye = torch.eye(
        3,
        dtype=F.dtype,
        device=F.device,
    ).unsqueeze(0)

    ortho_error = torch.max(
        torch.abs(
            R @ R.transpose(1, 2)
            - eye
        )
    )

    det_error = torch.max(
        torch.abs(
            torch.linalg.det(R)
            - 1.0
        )
    )

    if ortho_error >1e-5:
        raise RuntimeError(
            f"polar orthogonality error={ortho_error}"
        )

    if det_error >1e-5:
        raise RuntimeError(
            f"polar det error={det_error}"
        )

    return R
