log() {

```bash
from plyfile import PlyData
from pathlib import Path

candidates = list(Path('/data/wyh/RecycleGS/baselines').rglob('point_cloud.ply'))
print('找到 point_cloud.ply 数量:', len(candidates))
for p in candidates:
    ply = PlyData.read(str(p))
    names = ply['vertex'].data.dtype.names
    print('\\n', p)
    print(names)
    required = ['opacity','scale_0','scale_1','scale_2','rot_0','rot_1','rot_2','rot_3','f_dc_0','f_dc_1','f_dc_2']
    ok = all(x in names for x in required)
    print('is_gaussian_ply =', ok)
PY"
```
