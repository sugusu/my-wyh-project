# 1. 进入项目并设置日志

````bash
cd /data/wyh/RecycleGS
export CUDA_VISIBLE_DEVICES=2,3

log() {
  echo "" | tee -a /data/wyh/RecycleGS/realtime_log.md
  echo "### $(date '+%Y-%m-%d %H:%M:%S') $1" | tee -a /data/wyh/RecycleGS/realtime_log.md

  log "执行命令：$*"

# 10. 验证 TSGS Gaussian PLY 字段

执行：

```bash
run_cmd "python3 - <<'PY'
from plyfile import PlyData
from pathlib import Path

root = Path('/data/wyh/RecycleGS/baselines/tsgs_scene01')
candidates = list(root.rglob('point_cloud.ply'))
print('找到 point_cloud.ply 数量:', len(candidates))

required = [
    'opacity',
    'scale_0', 'scale_1', 'scale_2',
    'rot_0', 'rot_1', 'rot_2', 'rot_3',
    'f_dc_0', 'f_dc_1', 'f_dc_2'
]

for p in candidates:
    print('\\nPLY:', p)
    ply = PlyData.read(str(p))
    names = ply['vertex'].data.dtype.names
    print('字段:', names)
    ok = all(x in names for x in required)
    print('is_gaussian_ply =', ok)
PY"
```

必须确认：

```text
is_gaussian_ply = True
```
