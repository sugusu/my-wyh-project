# 4. 检查图像是否真的是 400 张独立图像

创建脚本：

```bash
cat > /data/wyh/RecycleGS/src/check_translab_full_data.py <<'PY'
from pathlib import Path
import hashlib
from PIL import Image
import json
from collections import Counter

scene = Path("/data/wyh/RecycleGS/data/translab_full/scene_01")
out = Path("/data/wyh/RecycleGS/outputs/debug")
out.mkdir(parents=True, exist_ok=True)

dirs = ["images", "delights", "masks", "normals", "transparent_masks"]
report = {}

def md5(path):
    h = hashlib.md5()
    with open(path, "rb") as f:
        h.update(f.read())
    return h.hexdigest()

for d in dirs:
    p = scene / d
    files = sorted([x for x in p.iterdir() if x.is_file()])
    hashes = []
    sizes = []
    image_shapes = []
    unreadable = []
    for f in files:
        try:
            hashes.append(md5(f))
            sizes.append(f.stat().st_size)
            with Image.open(f) as im:
                image_shapes.append([im.width, im.height, im.mode])
        except Exception as e:
            unreadable.append({"file": str(f), "error": str(e)})

    c = Counter(hashes)
    shape_counter = Counter([tuple(x) for x in image_shapes])
    report[d] = {
        "num_files": len(files),
        "num_unique_md5": len(set(hashes)),
        "num_duplicate_files": len(files) - len(set(hashes)),
        "min_size": min(sizes) if sizes else None,
        "max_size": max(sizes) if sizes else None,
        "mean_size": sum(sizes) / len(sizes) if sizes else None,
        "shapes": {str(k): v for k, v in shape_counter.items()},
        "num_unreadable": len(unreadable),
        "unreadable": unreadable[:10],
        "top_duplicate_hash_counts": c.most_common(5),
    }

with open(out / "translab_full_data_check.json", "w", encoding="utf-8") as f:
    json.dump(report, f, indent=2, ensure_ascii=False)

print(json.dumps(report, indent=2, ensure_ascii=False))
PY
```

运行：

```bash
run_cmd "python3 /data/wyh/RecycleGS/src/check_translab_full_data.py"
run_cmd "cat /data/wyh/RecycleGS/outputs/debug/translab_full_data_check.json"
```

重点看：

```text
images 的 num_unique_md5 是否接近 400
delights/normals/masks 是否接近 400
图像分辨率是否正常
```

如果 `images num_unique_md5` 很低，比如仍然只有 10，则停止训练并写报告。

# 9. 验证正式 Gaussian PLY

```bash
run_cmd "python3 - <<'PY'
from plyfile import PlyData
from pathlib import Path

root = Path('/data/wyh/RecycleGS/baselines/tsgs_scene01_full')
candidates = list(root.rglob('point_cloud.ply'))
print('找到 point_cloud.ply 数量:', len(candidates))

required = [
    'opacity',
    'scale_0', 'scale_1', 'scale_2',
    'rot_0', 'rot_1', 'rot_2', 'rot_3',
    'f_dc_0', 'f_dc_1', 'f_dc_2'
]

for p in candidates:
    print('\\nPLY:', p)
    ply = PlyData.read(str(p))
    names = ply['vertex'].data.dtype.names
    print('字段数量:', len(names))
    print('字段:', names)
    ok = all(x in names for x in required)
    print('is_gaussian_ply =', ok)
    print('文件大小MB:', p.stat().st_size / 1024 / 1024)
PY"
```

必须看到：
